@NM_SOU_DECLARATION_WAR_BY_STRONGS_DURATION_YEARS = 2

on_war_started = {
	on_actions = {
		# 宣戦布告された国とその周辺
		nm_nomads_fear_declaration_war_by_strongs
		# 宣戦布告した国の周辺
		nm_nomads_fear_strong_neighbor_has_started_war
	}
}

nm_nomads_fear_declaration_war_by_strongs = {
	trigger = {
		OR = {
			AND = {
				# 何らかの強大な敵に
				scope:attacker = {
					is_landed = yes
					is_independent_ruler = yes

					OR = {
						# 帝国級部族
						AND = {
							government_has_flag = government_is_tribal
							highest_held_title_tier = tier_empire
						}
						# 王国以上のnot部族
						AND = {
							NOT = { government_has_flag = government_is_tribal }
							highest_held_title_tier >= tier_kingdom
						}
					}
				}

				# 弱い部族が宣戦された
				scope:defender = {
					is_landed = yes
					is_independent_ruler = yes
					government_has_flag = government_is_tribal

					# 公爵あるいは伯爵
					highest_held_title_tier <= tier_duchy
				}
			}
		}
	}

	effect = {
		# 宣戦された人
		scope:defender = {
			if = {
				limit = {
					NOT = { has_character_flag = nm_sou_declaration_war_by_strongs }
					is_ai = yes
					is_landed = yes
					is_independent_ruler = yes
					government_has_flag = government_is_tribal
					# 公爵あるいは伯爵
					highest_held_title_tier <= tier_duchy
				}
				add_to_temporary_list = affectees
			}
		}
		# 宣戦された部族の近隣の部族
		scope:defender = {
			every_neighboring_and_across_water_top_liege_realm_owner = {
				limit = {
					NOT = { has_character_flag = nm_sou_declaration_war_by_strongs }
					is_ai = yes
					is_landed = yes
					is_independent_ruler = yes
					government_has_flag = government_is_tribal
					# 公爵あるいは伯爵
					highest_held_title_tier <= tier_duchy
				}
				add_to_temporary_list = affectees
			}
		}
		# 影響
		every_in_list = {
			list = affectees
			add_character_flag = {
				flag = nm_sou_declaration_war_by_strongs
				years = @NM_SOU_DECLARATION_WAR_BY_STRONGS_DURATION_YEARS
			}
		}
	}
}

nm_nomads_fear_strong_neighbor_has_started_war = {
	trigger = {
		# 開始した人
		scope:attacker = {
			is_landed = yes
			is_independent_ruler = yes

			OR = {
				# 帝国級部族
				AND = {
					government_has_flag = government_is_tribal
					highest_held_title_tier = tier_empire
				}
				# 王国以上のnot部族
				AND = {
					NOT = { government_has_flag = government_is_tribal }
					highest_held_title_tier >= tier_kingdom
				}
			}
		}

		# 開始した人の隣国
		scope:attacker = {
			any_neighboring_top_liege_realm_owner = {
				count > 0
				NOT = { has_character_flag = nm_sou_declaration_war_by_strongs }

				is_landed = yes
				is_independent_ruler = yes
				government_has_flag = government_is_tribal

				# 公爵あるいは伯爵
				highest_held_title_tier <= tier_duchy
			}
		}
	}
	effect = {
		scope:attacker = {
			every_neighboring_and_across_water_top_liege_realm_owner = {
				limit = {
					NOT = { has_character_flag = nm_sou_declaration_war_by_strongs }

					is_landed = yes
					is_independent_ruler = yes
					government_has_flag = government_is_tribal

					# 公爵あるいは伯爵
					highest_held_title_tier <= tier_duchy
				}
				add_to_temporary_list = affectees
			}
		}
		# 影響
		every_in_list = {
			list = affectees
			add_character_flag = {
				flag = nm_sou_declaration_war_by_strongs
				years = @NM_SOU_DECLARATION_WAR_BY_STRONGS_DURATION_YEARS
			}
		}
	}
}
